<div class="score">Score: <span id="score">0</span></div>
<canvas id="game" width="360" height="360"></canvas>

<div class="controls">
  <button class="up" onclick="setDir('UP')">⬆️</button>
  <button class="left" onclick="setDir('LEFT')">⬅️</button>
  <button class="down" onclick="setDir('DOWN')">⬇️</button>
  <button class="right" onclick="setDir('RIGHT')">➡️</button>
</div>

<script>
  const canvas = document.getElementById("game");
  const ctx = canvas.getContext("2d");
  const scoreEl = document.getElementById("score");

  const box = 20;
  let score = 0;

  let snake = [{ x: 9 * box, y: 9 * box }];
  let direction = "RIGHT";

  let food = {
    x: Math.floor(Math.random() * 18) * box,
    y: Math.floor(Math.random() * 18) * box
  };

  document.addEventListener("keydown", e => {
    if (e.key === "ArrowLeft") setDir("LEFT");
    if (e.key === "ArrowUp") setDir("UP");
    if (e.key === "ArrowRight") setDir("RIGHT");
    if (e.key === "ArrowDown") setDir("DOWN");
  });

  function setDir(dir) {
    if (dir === "LEFT" && direction !== "RIGHT") direction = "LEFT";
    if (dir === "UP" && direction !== "DOWN") direction = "UP";
    if (dir === "RIGHT" && direction !== "LEFT") direction = "RIGHT";
    if (dir === "DOWN" && direction !== "UP") direction = "DOWN";
  }

  function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    snake.forEach((s, i) => {
      ctx.fillStyle = i === 0 ? "#0f0" : "#7CFC00";
      ctx.fillRect(s.x, s.y, box, box);
    });

    ctx.fillStyle = "red";
    ctx.fillRect(food.x, food.y, box, box);

    let headX = snake[0].x;
    let headY = snake[0].y;

    if (direction === "LEFT") headX -= box;
    if (direction === "UP") headY -= box;
    if (direction === "RIGHT") headX += box;
    if (direction === "DOWN") headY += box;

    if (headX === food.x && headY === food.y) {
      score++;
      scoreEl.textContent = score;
      food = {
        x: Math.floor(Math.random() * 18) * box,
        y: Math.floor(Math.random() * 18) * box
      };
    } else {
      snake.pop();
    }

    const newHead = { x: headX, y: headY };

    if (
      headX < 0 || headY < 0 ||
      headX >= canvas.width || headY >= canvas.height ||
      snake.some(p => p.x === newHead.x && p.y === newHead.y)
    ) {
      clearInterval(game);
      alert("Game Over! Score: " + score);
    }

    snake.unshift(newHead);
  }

  const game = setInterval(draw, 120);
</script>
